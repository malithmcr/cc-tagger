## Install

```bash
npm install --save react-cc-tagger
```

## Usage

```jsx
import React, { Component } from 'react'

import ccTagger from 'react-cc-tagger'

class Example extends Component {
  render () {
    return (
      <ccTagger />
    )
  }
}
```
